SELECT IL.NAME, I.DESCRIPTION  FROM issueslists IL
JOIN issues i on IL.ID = i.ISSUESLIST_ID;

SELECT  U.FIRSTNAME, U.LASTNAME ,I.SUMMARY FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS SUM_TASKS FROM users U
JOIN issues i on U.ID = i.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME;
